ROC curve (AUC-ROC)

Code for your evaluator.py

src/training/evaluator.py

from sklearn.metrics import (
    classification_report,
    roc_auc_score,
    average_precision_score,
)

def evaluate(y_true, y_pred, y_prob):
    """
    NEVER use accuracy for fraud.
    Use: precision, recall, F1, AUC-PR.
    """
    print(classification_report(
        y_true, y_pred,
        target_names=['legit', 'fraud']
    ))
    auc_roc = roc_auc_score(y_true, y_prob)
    auc_pr  = average_precision_score(y_true, y_prob)
    print(f"AUC-ROC : {auc_roc:.4f}")
    print(f"AUC-PR  : {auc_pr:.4f}  ← more important!")
    return {"auc_roc": auc_roc, "auc_pr": auc_pr}

Decision rule: Start with class_weight='balanced'. If recall is still too low AND you have <100k samples, add SMOTE inside a pipeline. Never SMOTE before your train/test split — that's data leakage.

The right way — SMOTE inside CV

src/training/trainer.py

from imblearn.over_sampling import SMOTE
from imblearn.pipeline import Pipeline as ImbPipeline
from sklearn.preprocessing import StandardScaler
from sklearn.model_selection import StratifiedKFold
from sklearn.metrics import average_precision_score
import xgboost as xgb

# WRONG: leaks test data into synthetic samples
# X_res, y_res = SMOTE().fit_resample(X, y)
# X_train, X_test = train_test_split(X_res, y_res)

# RIGHT: pipeline applies SMOTE per fold only
pipeline = ImbPipeline([
    ('scaler', StandardScaler()),
    ('smote', SMOTE(
        sampling_strategy=0.1,  # fraud → 10% of total
        random_state=42
    )),
    ('model', xgb.XGBClassifier(
        scale_pos_weight=99,   # fallback weight too
        eval_metric='aucpr',
        random_state=42
    ))
])

cv = StratifiedKFold(n_splits=5, shuffle=True, random_state=42)
scores = []
for fold, (tr_idx, val_idx) in enumerate(cv.split(X, y)):
    pipeline.fit(X[tr_idx], y[tr_idx])
    # SMOTE only saw X[tr_idx] — no leakage ✓
    probs = pipeline.predict_proba(X[val_idx])[:,1]
    score = average_precision_score(y[val_idx], probs)
    scores.append(score)
    print(f"Fold {fold} AUC-PR: {score:.4f}")

print(f"Mean AUC-PR: {sum(scores)/len(scores):.4f}")

results = []

Threshold tuning code

import numpy as np
from sklearn.metrics import precision_recall_curve

def find_threshold(y_true, y_prob, target_recall=0.90):
    """
    Find lowest threshold that hits 90% recall.
    In fraud: catching 90% of fraud is usually
    the minimum requirement.
    """
    precision, recall, thresholds = \
        precision_recall_curve(y_true, y_prob)

    # Find all thresholds at target recall
    valid = thresholds[recall[:-1] >= target_recall]

    if len(valid) == 0:
        return 0.10  # can't hit target, use low threshold

    # Highest threshold = best precision at that recall
    optimal = valid.max()

    idx = np.argmin(np.abs(thresholds - optimal))
    print(f"Threshold : {optimal:.3f}")
    print(f"Precision : {precision[idx]:.3f}")
    print(f"Recall    : {recall[idx]:.3f}")
    return optimal

src/api/routes/predict.py — complete implementation

from fastapi import APIRouter, Depends, HTTPException
from src.core.schemas import TransactionRequest, PredictionResponse
from src.features.engineer import FeatureEngineer
from src.models.registry import ModelRegistry
import time, uuid

router = APIRouter()

@router.post("/predict", response_model=PredictionResponse)
async def predict(
    transaction: TransactionRequest,
    model_registry: ModelRegistry = Depends(),
    feature_eng: FeatureEngineer = Depends()
):
    start_ms = time.perf_counter() * 1000

    # 1. Extract features from Redis feature store
    try:
        features = await feature_eng.extract(transaction)
    except Exception as e:
        raise HTTPException(502, f"Feature extraction failed: {e}")

    # 2. Get current production model from MLflow registry
    model = model_registry.get_production_model()

    # 3. Fraud probability
    fraud_prob = float(
        model.predict_proba(features.reshape(1, -1))[0, 1]
    )

    # 4. Apply threshold (tune per business need — see Phase 5)
    FRAUD_THRESHOLD = 0.35  # targets ~90% recall
    is_fraud = fraud_prob >= FRAUD_THRESHOLD

    # 5. Risk tier
    risk_tier = (
        "HIGH"   if fraud_prob >= 0.70 else
        "MEDIUM" if fraud_prob >= 0.35 else
        "LOW"
    )

    latency = time.perf_counter() * 1000 - start_ms

    return PredictionResponse(
        transaction_id=transaction.transaction_id,
        fraud_probability=round(fraud_prob, 4),
        fraud_label=is_fraud,
        risk_tier=risk_tier,
        model_version=model.version,
        latency_ms=round(latency, 1),
        decision_id=f"dec_{uuid.uuid4().hex[:8]}"
    )

src/core/schemas.py

from pydantic import BaseModel, Field
from datetime import datetime
from typing import Optional

class TransactionRequest(BaseModel):
    transaction_id: str
    user_id: str
    amount: float = Field(..., gt=0)
    merchant_id: str
    merchant_category: str
    timestamp: datetime
    currency: str = "USD"
    device_fingerprint: Optional[str] = None
    ip_address: Optional[str] = None
    card_present: bool = False
    payment_method: str

class PredictionResponse(BaseModel):
    transaction_id: str
    fraud_probability: float
    fraud_label: bool
    risk_tier: str  # LOW | MEDIUM | HIGH
    model_version: str
    latency_ms: float
    decision_id: str
